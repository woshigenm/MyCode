sbit beep = P2^5;
unsigned char KeyNum = 0;
unsigned char FreqSelect = 0;

void main()
{
	unsigned int delayTime;
	
	unsigned int currentIndex = 0; // 当前音符索引
	ScaleNote * currentNote = Song;
	
	LCD_Init();
	LCD_ShowString(1, 1, "FUCK YOU FAN");
	Timer0_Init();
	
	while(1)
	{
		if(currentNote->noteIndex != END_MARKER)
		{
			if(currentNote->noteIndex != SILENCE) {
				FreqSelect = currentNote->noteIndex;
                TR0 = 1; // 启动定时器
            }
            
            // 计算实际时值（去除标志位）
            delayTime = SPEED / 4 * (currentNote->duration & 0x3F);
            
            // 音符持续时间
            Delay(delayTime);
            
            // 处理圆滑线标志
            if(!(currentNote->duration & SLUR)) {
                TR0 = 0; // 停止定时器
                beep = 0; // 确保蜂鸣器关闭
                Delay(5); // 音符间停顿
				TR0 = 1;
            }
			
			currentNote++;
		}else
		{	
			TR0 = 0;
			beep = 0;
			
		}
	}
}

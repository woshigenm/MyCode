sbit beep = P2^5;
unsigned char KeyNum = 0;
unsigned char FreqSelect = 0;

void main()
{
	unsigned int delayTime;
	
	unsigned int currentIndex = 0; // 当前音符索引
	ScaleNote * currentNote = Song;
	
	LCD_Init();
	LCD_ShowString(1, 1, "FUCK YOU FAN");
	Timer0_Init();
	
	while(1)
	{
		if(currentNote->noteIndex != END_MARKER)
		{
			if(currentNote->noteIndex != SILENCE) {
				FreqSelect = currentNote->noteIndex;
                TR0 = 1; // 启动定时器
            }
            
            // 计算实际时值（去除标志位）
            delayTime = SPEED / 4 * (currentNote->duration & 0x3F);
            
            // 音符持续时间
            Delay(delayTime);
            
            // 处理圆滑线标志
            if(!(currentNote->duration & SLUR)) {
                TR0 = 0; // 停止定时器
                beep = 0; // 确保蜂鸣器关闭
                Delay(5); // 音符间停顿
				TR0 = 1;
            }
			
			currentNote++;
		}else

// 辅助函数：比较两个链表是否相等
bool CompareLists(LinkList L1, LinkList L2) {
	if (ListLength(L1) != ListLength(L2)) return false;

	Node *p1 = L1->next;
	Node *p2 = L2->next;

	while (p1 && p2) {
		if (p1->data != p2->data) return false;
		p1 = p1->next;
		p2 = p2->next;
	}

	return (p1 == NULL && p2 == NULL);
}

// 辅助函数：生成随机链表
Status CreateRandomList(LinkList *L, int n, int min, int max) {
	if (!L || n <= 0) return ERROR;

	if (CreateList_R(L, 0) == ERROR) return ERROR;  // 创建空链表

	srand(time(NULL));
	for (int i = 0; i < n; i++) {
		ElemType val = min + rand() % (max - min + 1);
		if (ListInsert(*L, i + 1, val) == ERROR) {
			DestroyList(L);
			return ERROR;
		}
	}
	return OK;
}

		GetElem(L, random_pos, &e);
	}
	end = clock();
	cpu_time_used = ((double) (end - start)) / CLOCKS_PER_SEC;
	printf("  1000次随机访问耗时: %.6f秒\n", cpu_time_used);

	// 测试9.3：随机插入性能
	printf("9.3 随机插入性能测试:\n");
	start = clock();
	for (int i = 0; i < 1000; i++) {
		random_pos = 1 + rand() % (ListLength(L) + 1);
		ListInsert(L, random_pos, i + 10000);
	}
	end = clock();
	cpu_time_used = ((double) (end - start)) / CLOCKS_PER_SEC;
	printf("  1000次随机插入耗时: %.6f秒\n", cpu_time_used);
	printf("  插入后链表长度: %d\n", ListLength(L));

	// 测试9.4：随机删除性能
	printf("9.4 随机删除性能测试:\n");
	start = clock();
	for (int i = 0; i < 1000; i++) {
		random_pos = 1 + rand() % ListLength(L);
		ListDelete(L, random_pos, &e);
	}
	end = clock();
	cpu_time_used = ((double) (end - start)) / CLOCKS_PER_SEC;
	printf("  1000次随机删除耗时: %.6f秒\n", cpu_time_used);
	printf("  删除后链表长度: %d\n", ListLength(L));

	DestroyList(&L);
	printf("\n");
}

// 测试用例10：综合测试
void test_integration() {
	printf("===== 测试用例10：综合测试 =====\n");

	LinkList L = NULL;

	// 步骤1：创建链表
	printf("步骤1: 创建包含10个元素的链表\n");
	if (CreateList_R(&L, 10) == OK) {
		ListPrint(L);
	}

	// 步骤2：插入元素
	printf("\n步骤2: 在多个位置插入元素\n");
	ListInsert(L, 1, 100);   // 头部插入
	ListInsert(L, 5, 200);   // 中间插入
	int len = ListLength(L);
	ListInsert(L, len, 300); // 尾部插入
	ListPrint(L);

	// 步骤3：删除元素
	printf("\n步骤3: 删除多个位置的元素\n");
	ElemType deleted;
	ListDelete(L, 1, &deleted);
	printf("删除头部元素: %d\n", deleted);

	len = ListLength(L);
	ListDelete(L, len, &deleted);
	printf("删除尾部元素: %d\n", deleted);

	ListDelete(L, 3, &deleted);
	printf("删除中间元素: %d\n", deleted);

	ListPrint(L);

	// 步骤4：查找操作
	printf("\n步骤4: 查找操作\n");
	Node *node = NULL;
	if (LocateElem(L, 5, &node) == OK) {
		printf("找到元素5, 地址: %p\n", node);
	}

	Node *pre = NULL, *next = NULL;
	if (PriorElem(L, 5, &pre) == OK) {
		printf("元素5的前驱: %d\n", pre->data);
	}

	if (NextElem(L, 5, &next) == OK) {
		printf("元素5的后继: %d\n", next->data);
	}

	// 步骤5：使用PriorInsert和NextInsert
	printf("\n步骤5: 使用PriorInsert和NextInsert\n");
	PriorInsert(L, 2, 500);
	printf("在第2个节点前插入500\n");

	NextInsert(L, 4, 600);
	printf("在第4个节点后插入600\n");

	ListPrint(L);

	// 步骤6：清空链表
	printf("\n步骤6: 清空链表\n");
	ClearList(L);
	printf("清空后链表长度: %d\n", ListLength(L));
	printf("链表是否为空: %s\n", ListEmpty(L) ? "是" : "否");

	// 步骤7：重新填充
	printf("\n步骤7: 重新填充链表\n");
	for (int i = 1; i <= 5; i++) {
		ListInsert(L, i, i * 1000);
	}
	ListPrint(L);

	// 步骤8：最终销毁
	printf("\n步骤8: 销毁链表\n");
	DestroyList(&L);
	printf("链表指针: %p\n", L);

	printf("\n");
}
